import { useState } from 'react'

export const Counter = ({ initialCount = 0 }) => {
  const [count, setCount] = useState(initialCount)

  return (
    <div style={{ 
      fontFamily: 'Arial, sans-serif',
      maxWidth: '300px',
      margin: '20px auto',
      padding: '20px',
      border: '2px solid #ccc',
      borderRadius: '10px',
      textAlign: 'center'
    }}>
      <h2>Counter: {count}</h2>
      <div style={{ display: 'flex', justifyContent: 'center', gap: '10px', marginTop: '20px' }}>
        <button onClick={() => setCount(c => c - 1)} style={buttonStyle}>-</button>
        <button onClick={() => setCount(initialCount)} style={buttonStyle}>Reset</button>
        <button onClick={() => setCount(c => c + 1)} style={buttonStyle}>+</button>
      </div>
    </div>
  )
}

export const buttonStyle = {
  padding: '10px 20px',
  fontSize: '16px',
  cursor: 'pointer',
  backgroundColor: '#4CAF50',
  color: 'white',
  border: 'none',
  borderRadius: '5px',
  transition: 'background-color 0.3s'
}
